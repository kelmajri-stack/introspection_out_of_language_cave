function generateArt() {
    const text = document.getElementById("textInput").value;
    const languages = detectLanguages(text);

    generateVisualArt(languages);
    generateMusic(languages);
}

function detectLanguages(text) {
    const hasArabic = /[\u0600-\u06FF]/.test(text);
    const hasEnglish = /[A-Za-z]/.test(text);
    return { arabic: hasArabic, english: hasEnglish };
}

function generateVisualArt(langs) {
    const canvas = document.getElementById("artCanvas");
    const ctx = canvas.getContext("2d");
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    if (langs.arabic) {
        ctx.fillStyle = "blue";
        ctx.fillRect(50, 50, 100, 100);
    }

    if (langs.english) {
        ctx.fillStyle = "red";
        ctx.fillRect(150, 50, 100, 100);
    }
}

function generateMusic(langs) {
    const audioElement = document.getElementById("audioOutput");
    audioElement.src = "https://example.com/sample.mp3"; // سيتم استبدالها لاحقًا
    alert("سيتم توليد موسيقى بناءً على اللغات: " + JSON.stringify(langs));
}
